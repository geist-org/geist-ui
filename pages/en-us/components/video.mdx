## Video

Display video content.

<Playground
  scope={{ Video }}
  code={`
<Video
  src="/videos/big-buck-bunny.mp4"
  poster="/images/big-buck-bunny-poster.jpg"
  width={2000 / 3.5}
  height={1000 / 3.5}
/>`} />

<Playground
  title="Disable fullscreen"
  scope={{ Video }}
  code={`
<Video
  src="/videos/big-buck-bunny.mp4"
  poster="/images/big-buck-bunny-poster.jpg"
  width={2000 / 3.5}
  height={1000 / 3.5}
  fullscreenable={false}
/>`} />

<Playground
  title="Loop"
  scope={{ Video }}
  code={`
<Video
  src="/videos/big-buck-bunny.mp4"
  poster="/images/big-buck-bunny-poster.jpg"
  width={2000 / 3.5}
  height={1000 / 3.5}
  loop
/>`} />

<Attributes edit="/pages/en-us/components/video.mdx">
<Attributes.Title>Video.Props</Attributes.Title>

| Attribute | Description | Type | Accepted values | Default
| ---------- | ---------- | ---- |  -------------- | ------ |
| **src** | Video src | `string` | - | - |
| **width** | Video width | `number` | - | - |
| **height** | Video height | `number` | - | - |
| **fullscreenable** | Allow fullscreen | `boolean` | - | `ture` |
| ... | native props | `VideoHTMLAttributes` | `'mute', 'loop', 'className', ...` | - |

</Attributes>

export default ({ children }) => <Layout meta={meta}>{children}</Layout>
